func (self *ZipEntry) readClass(className string) ([]byte, Entry, error) {
	r, err := zip.OpenReader(self.absPath)

		return nil, nil, err

			}

			data, err := ioutil.ReadAll(rc)
			if err != nil {

			}

			return data, self, nil
		}
	}

}
